# Color 2
class Color2 {
    private int r;
    private int g;
    private int b;


    Color2(int r, int g, int b) {
        this.r = r;
        this.g = g;
        this.b = b;

    }

    int getR() {
        return r;
    }

    int getG() {
        return g;
    }

    int getB() {
        return b;
    }


}
